from aiogram import Bot, Dispatcher, types from aiogram.types import ReplyKeyboardMarkup, KeyboardButton from aiogram.utils import executor import json TOKEN = "TON_TOKEN_ICI" bot = Bot(token=TOKEN) dp = Dispatcher(bot) # Menu principal menu = ReplyKeyboardMarkup(resize_keyboard=True) menu.add(KeyboardButton("📚 Chapitres")) menu.add(KeyboardButton("📊 Mon niveau")) # Charger utilisateurs def load_users(): try: with open("data/users.json", "r") as f: return json.load(f) except: return {} def save_users(data): with open("data/users.json", "w") as f: json.dump(data, f) @dp.message_handler(commands=['start']) async def start(msg: types.Message): users = load_users() user_id = str(msg.from_user.id) if user_id not in users: users[user_id] = {"level": 1} save_users(users) await msg.answer("Bienvenue dans la formation 📱🔧", reply_markup=menu) @dp.message_handler(lambda message: message.text == "📊 Mon niveau") async def level(msg: types.Message): users = load_users() user_id = str(msg.from_user.id) lvl = users[user_id]["level"] await msg.answer(f"Ton niveau actuel : {lvl}") @dp.message_handler(lambda message: message.text == "📚 Chapitres") async def chapters(msg: types.Message): text = """ 📚 Chapitre 1 : Multimètre Niveau 1 débloqué 📚 Chapitre 2 : Composants 🔒 Bloqué """ await msg.answer(text) if __name__ == "__main__": executor.start_polling(dp)
